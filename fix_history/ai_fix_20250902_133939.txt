def multiply_numbers(a: float, b: float) -> float:
    """Multiply two numbers and return the result."""
    return a * b


def divide_numbers(a: float, b: float) -> float:
    """Divide two numbers and return the result. Raise an Exception if division by zero is attempted."""
    if b == 0:
        raise ValueError("Cannot divide by zero.")  # Raise Exception instead of returning string
    return a / b
